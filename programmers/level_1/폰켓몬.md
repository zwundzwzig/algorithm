import java.util.HashMap;

class Solution {
    public int solution(int[] nums) {
        int mons = nums.length / 2;
        
        HashMap<Integer, Integer> map = new HashMap<>();
        
        for (int num : nums) 
            map.put(num, map.getOrDefault(num, 0) + 1);
        
        return (map.keySet().size() >= mons) ? mons : map.keySet().size();
    }
}
